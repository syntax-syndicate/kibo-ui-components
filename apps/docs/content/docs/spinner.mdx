<Preview name="spinner" code={`'use client';

const variants = ['default', 'circle', 'pinwheel', 'circle-filled', 'ellipsis', 'ring', 'bars', 'infinite'];

  <div className="grid grid-cols-4 w-full h-screen items-center justify-center gap-8 bg-secondary">

  </div>
);

export default Example;`} />
